# Framework Guide -- Vue

Integrate the Elevate Design System into Vue 3 projects. This guide covers Vite + Vue 3 and Nuxt 3 setups with Composition API.

---

## Prerequisites

- Node.js >= 22.0.0
- npm >= 9.0.0

```bash
npm install git+https://github.com/sqli/elevate-design-system.git
```

---

## Setup with Vite + Vue 3

### 1. Create the project

```bash
npm create vite@latest my-app -- --template vue-ts
cd my-app
npm install
npm install git+https://github.com/sqli/elevate-design-system.git
```

### 2. Import CSS

**Option A: In your CSS file**

```css
/* src/style.css */
@import '@sqli/elevate-design-system';
@import '@sqli/elevate-design-system/fonts';
```

**Option B: In your JS entry point**

```ts
// src/main.ts
import '@sqli/elevate-design-system';
import '@sqli/elevate-design-system/fonts';
import { createApp } from 'vue';
import App from './App.vue';

createApp(App).mount('#app');
```

### 3. Configure the theme

Set the `data-theme` attribute on the `<html>` element in `index.html`:

```html
<!doctype html>
<html lang="en" data-theme="sqli-light">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>My App</title>
  </head>
  <body>
    <div id="app"></div>
    <script type="module" src="/src/main.ts"></script>
  </body>
</html>
```

## Setup with Nuxt 3

### 1. Create the project

```bash
npx nuxi@latest init my-app
cd my-app
npm install
npm install git+https://github.com/sqli/elevate-design-system.git
```

### 2. Configure nuxt.config.ts

```ts
// nuxt.config.ts
export default defineNuxtConfig({
  css: [
    '@sqli/elevate-design-system',
    '@sqli/elevate-design-system/fonts',
  ],
  app: {
    head: {
      htmlAttrs: {
        'data-theme': 'sqli-light',
        lang: 'en',
      },
    },
  },
});
```

With this configuration, Nuxt handles the CSS imports and HTML attributes automatically. No additional setup is needed.

### 3. Verify the setup

Create a test page to confirm the design system is loaded:

```vue
<!-- pages/index.vue -->
<template>
  <div class="min-h-screen bg-base-100 text-base-content p-8">
    <h1 class="text-3xl font-bold mb-4">Elevate Design System</h1>
    <button class="btn btn-primary">It works</button>
  </div>
</template>
```

## Wrapper Components

The design system is CSS-only. You apply DaisyUI class names directly in your templates. The following wrapper components provide a typed API using `defineProps` for the most common patterns.

### EButton

```vue
<!-- src/components/EButton.vue -->
<script setup lang="ts">
type ButtonVariant = 'primary' | 'secondary' | 'accent' | 'ghost' | 'link' | 'outline';
type ButtonSize = 'lg' | 'md' | 'sm' | 'xs';

const props = withDefaults(
  defineProps<{
    variant?: ButtonVariant;
    size?: ButtonSize;
    loading?: boolean;
    wide?: boolean;
    disabled?: boolean;
  }>(),
  {
    variant: 'primary',
    size: 'md',
    loading: false,
    wide: false,
    disabled: false,
  }
);

const variantClasses: Record<ButtonVariant, string> = {
  primary: 'btn-primary',
  secondary: 'btn-secondary',
  accent: 'btn-accent',
  ghost: 'btn-ghost',
  link: 'btn-link',
  outline: 'btn-outline',
};

const sizeClasses: Record<ButtonSize, string> = {
  lg: 'btn-lg',
  md: 'btn-md',
  sm: 'btn-sm',
  xs: 'btn-xs',
};
</script>

<template>
  <button
    class="btn"
    :class="[
      variantClasses[props.variant],
      sizeClasses[props.size],
      { 'loading': props.loading, 'btn-wide': props.wide },
    ]"
    :disabled="props.disabled || props.loading"
  >
    <slot />
  </button>
</template>
```

**Usage:**

```vue
<EButton variant="primary" size="lg">Save Changes</EButton>
<EButton variant="ghost" size="sm">Cancel</EButton>
<EButton variant="outline" :loading="true">Submitting</EButton>
```

### ECard

```vue

<script setup lang="ts">

const props = withDefaults(
  defineProps<{
    title?: string;

    variant?: CardVariant;
  }>(),
  {
    title: undefined,
    image: undefined,
    imageAlt: '',
    variant: 'default',
  }

<template>
  <div
    class="card bg-base-100 shadow-xl"
    :class="{
      'card-bordered': props.variant === 'bordered',
      'card-compact': props.variant === 'compact',

    </figure>

      <slot />
      <div v-if="$slots.actions" class="card-actions justify-end">
        <slot name="actions" />
      </div>
    </div>
  </div>
</template>
```

```vue
</ECard>
```

## Theme Switching with Vue Composable

### useTheme Composable

```ts
// src/composables/useTheme.ts
import { ref, watch, onMounted } from 'vue';

type Theme = 'sqli-light' | 'sqli-dark';

const STORAGE_KEY = 'elevate-theme';

const theme = ref<Theme>('sqli-light');

function applyTheme(value: Theme) {
  document.documentElement.setAttribute('data-theme', value);
  localStorage.setItem(STORAGE_KEY, value);
}

function getInitialTheme(): Theme {
  const stored = localStorage.getItem(STORAGE_KEY);
  if (stored === 'sqli-light' || stored === 'sqli-dark') return stored;

  const prefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches;
  return prefersDark ? 'sqli-dark' : 'sqli-light';
}

export function useTheme() {
  onMounted(() => {
    theme.value = getInitialTheme();
    applyTheme(theme.value);

    const mediaQuery = window.matchMedia('(prefers-color-scheme: dark)');
    mediaQuery.addEventListener('change', (e) => {
      const stored = localStorage.getItem(STORAGE_KEY);
      if (!stored) {
        theme.value = e.matches ? 'sqli-dark' : 'sqli-light';
      }
    });
  });

  watch(theme, (newTheme) => {
    applyTheme(newTheme);
  });

  function toggleTheme() {
    theme.value = theme.value === 'sqli-light' ? 'sqli-dark' : 'sqli-light';
  }

  function setTheme(value: Theme) {
    theme.value = value;
  }

  const isDark = ref(false);
  watch(theme, (val) => {
    isDark.value = val === 'sqli-dark';
  }, { immediate: true });

  return { theme, isDark, toggleTheme, setTheme };
}
```

### Nuxt 3: useTheme as a Nuxt Composable

For Nuxt 3, place the composable in the `composables/` directory for auto-import:

```ts
// composables/useTheme.ts
export function useTheme() {
  type Theme = 'sqli-light' | 'sqli-dark';
  const STORAGE_KEY = 'elevate-theme';

  const theme = useState<Theme>('elevate-theme', () => 'sqli-light');

  function applyTheme(value: Theme) {
    if (import.meta.client) {
      document.documentElement.setAttribute('data-theme', value);
      localStorage.setItem(STORAGE_KEY, value);
    }
  }

  onMounted(() => {
    const stored = localStorage.getItem(STORAGE_KEY);
    if (stored === 'sqli-light' || stored === 'sqli-dark') {
      theme.value = stored;
    } else {
      const prefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches;
      theme.value = prefersDark ? 'sqli-dark' : 'sqli-light';
    }
    applyTheme(theme.value);
  });

  watch(theme, (newTheme) => applyTheme(newTheme));

  function toggleTheme() {
    theme.value = theme.value === 'sqli-light' ? 'sqli-dark' : 'sqli-light';
  }

  const isDark = computed(() => theme.value === 'sqli-dark');

  return { theme, isDark, toggleTheme };
}
```

## Using the Tailwind Preset

If your Vue project uses Tailwind CSS 4, you can extend it with SQLI brand tokens.

```bash
npm install -D tailwindcss@^4.1.0
```

```javascript
// tailwind.config.js
import sqliPreset from '@sqli/elevate-design-system/preset';

export default {
  presets: [sqliPreset],
  content: ['./index.html', './src/**/*.{vue,js,ts}'],
};
```

For Nuxt 3 with the `@nuxtjs/tailwindcss` module:

```ts
// tailwind.config.ts
import sqliPreset from '@sqli/elevate-design-system/preset';

export default {
  presets: [sqliPreset],
  content: [],
};
```

This adds the following utilities to your Tailwind classes:

| Utility | Example Class | Value |
|---------|---------------|-------|
| Colors | `bg-sqli-cobalt`, `text-sqli-cream` | SQLI brand colors |
| Font Family | `font-sans`, `font-everett` | TWK Everett |
| Border Radius | `rounded-sqli-sm`, `rounded-sqli-lg` | Brand-consistent rounding |
| Shadows | `shadow-sqli-md` | Brand shadows |
| Animations | `animate-sqli-fade-in` | Brand transitions |

```vue
<template>
  <div class="bg-sqli-cream text-sqli-midnight font-everett">
    <h1 class="text-sqli-cobalt">Branded Heading</h1>
    <div class="rounded-sqli-lg shadow-sqli-md p-6">
      <button class="btn btn-primary">DaisyUI Button</button>
    </div>
  </div>
</template>
```
